page 的 demo

完全版分页

S.use('kg/page/2.0.2/index', function (S, page) {

var pageObj = new page({

container: "#page1", // 容器

total_page: 11, // 总页数

continuous_page: 5, // 连续显示分页数

current_page: 6, // 当前页

first_page: rue, // 首页

preview_show: true, // 是否显示第一页

next_show: true, // 是否显示下一页

last_show: true, // 最后一页

edge_page: 2, // 两端显示分页数

skip_show: true // 是否显示跳转

});

pageObj.on("page:skip", function(e){

alert(e.pageNum);

});

});
